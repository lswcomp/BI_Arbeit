BigQuery-Abfrage:
\begin{verbatim}    
#standardSQL
SELECT date, client, pct_urls, sample_urls
FROM `httparchive.blink_features.usage`
WHERE feature = 'WebAssemblyInstantiation'
AND date = (
  SELECT MAX(date) 
  FROM `httparchive.blink_features.usage`
)
ORDER BY date DESC, client
\end{verbatim}
